  // 1: 改变原数组 forEach(), push(), pop(), unshift(), shift(), splice()


  // 2: 不改变原数组 map(), filter(), every(), some(), reduce(), concat(), slice()


  // 3: 判断一个对象是否为数组: Array.isArray()


  // 4: 将数组转为对象的方法: toString(), join()

  /**
   * 1. Array.map()
   * 此方法是将数组中的每个元素调用一个提供的函数，结果作为一个新的数组返回，并没有改变原来的数组
   *
  */
  let arr = [1, 2, 3, 4, 5];
  let newArr = arr.map(item => item * 2);
  console.log('原数组', arr); // 原数组保持不变 [1, 2, 3, 4, 5]
  console.log('返回的新数组', newArr); // [2, 4, 6, 8, 10]

 /**
   * 2. Array.forEach()
   * 此方法是将数组中的每个元素执行传进提供的函数，没有返回值，直接改变原数组，注意和 map 方法区分
   *
  */
  let arr1 = [1, 2, 3, 4, 5];
  arr1.forEach((item, index) => arr1[index] = item * 2)
  console.log('原数组', arr1); // 改变原数组 [2, 4, 6, 8, 10]

  /**
   * 3. Array.filter()
   * 此方法是将所有元素进行判断，将满足条件的元素作为一个新的数组返回,返回值是一个数组
   *
  */
  let arr2 = [1, 2, 3, 4, 5];
  var newArr2 = arr2.filter(item => item > 2)
  console.log('原数组', arr2); // 原数组保持不变 [1, 2, 3, 4, 5]
  console.log('返回的新数组', newArr2); // [3, 4, 5]

  /**
   * 4. Array.every()
   * 此方法是将所有元素进行判断返回一个布尔值，如果所有元素都满足判断条件，则返回 true，否则为 false：返回值是boolean
   *
  */
  let arr3 = [1, 2, 3, 4, 5];
  const isLessTan4 = value => value < 4;
  const isLessTan6 = value => value < 6;
  let result1 = arr3.every(isLessTan4);
  let result2 = arr3.every(isLessTan6);
  console.log(result1);
  console.log(result2);
  console.log('原数组', arr3); // 原数组保持不变 [1, 2, 3, 4, 5]

  /**
   * 5. Array.some()
   * 此方法是将所有元素进行判断返回一个布尔值，如果有一个元素满足条件，则返回true , 剩余的元素不会再执行检测。若所有元素都不满足判断条件，则返回 false：
   *
  */
  let arr4 = [1, 2, 3, 4, 5];
  const lessTan4 = value => value < 4;
  const lessTan6 = value => value > 6;
  let res1 = arr4.some(lessTan4);
  let res2 = arr4.some(lessTan6);
  console.log(res1);
  console.log(res2);
  console.log('原数组', arr4); // [1, 2, 3, 4, 5]

  /**
   * 6. Array.reduce()
   * 此方法是所有元素调用返回函数，返回值为最后结果, 传入的值必须是函数类型：用于求和
   *
  */
  let arr5 = [1, 2, 3, 4, 5];
  let sum = arr5.reduce((preValue, num) => preValue + num, 0) // 第二个参数为sum的初始值
  console.log('总和', sum);
  console.log('原数组', arr5); // 原数组保持不变 [1, 2, 3, 4, 5]

  /**
   * 7. Array.push()
   * 此方法是在数组的后面添加新加元素，此方法改变了原数组,并返回数组长度
   *
  */
  let arr6 = [1, 2, 3, 4, 5];
  let length = arr6.push(6)
  console.log('返回值为数组长度', length);
  console.log('原数组', arr6); // 改变原数组 [1, 2, 3, 4, 5, 6]

  /**
   * 8. Array.pop()
   * 此方法在数组后面删除最后一个元素，此方法改变了原数组，返回被移除的数组元素
   *
  */
  let arr7 = ['broccoli', 'cauliflower', 'cabbage', 'kale', 'tomato'];
  let item = arr7.pop();
  console.log('返回值为被移除的数组元素', item); // tomato
  console.log('原数组', arr7); // 改变原数组 ['broccoli', 'cauliflower', 'cabbage', 'kale']

  /**
   * 9. Array.unshift()
   * 此方法是将一个或多个元素添加到数组的开头，并返回新数组的长度：
   *
  */
  let arr9 = ['broccoli', 'cauliflower', 'cabbage', 'kale', 'tomato'];
  let len = arr9.unshift('1', '2')
  console.log('返回值为新数组长度', len); // 7
  console.log('原数组', arr9); // [1, 2, 'broccoli', 'cauliflower', 'cabbage', 'kale', 'tomato']

  /**
   * 10. Array.shift()
   * 此方法在删除数组开头的第一个元素，此方法改变了原数组，返回被移除的数组元素
   *
  */
  let arr8 = ['broccoli', 'cauliflower', 'cabbage', 'kale', 'tomato'];
  let data = arr8.shift();
  console.log('返回值为被移除的数组元素', data); // broccoli
  console.log('原数组', arr8); // ['broccoli', 'cauliflower', 'cabbage', 'kale']

  /**
   * 11. Array.splice(开始位置,删除个数,元素)
   * 万能的方法,用于数组的增删改,splice()方法实际是一个删除数组元素方法，但可以拓展出插入，和替换两个用法
   * 返回值接收的是被删除的元素
  */
  let test = [1, 2, 3, 4, 5];
  let test1 = test.splice(2, 0, 'hello')
  console.log(test1) //[1, 2, 'hello', 3, 4, 5]新增一个元素
  console.log('test', test);
  let test2 = test.splice(2, 3)
  console.log(test2) //[1, 2, 5] 删除三个元素
  console.log('test', test);
  let test3 = test.splice(2, 1, 'hello')
  console.log(test3) //[1, 2, 'hello'] 替换一个元素
  console.log('test', test);

  /**
   * 12. Array.slice(begin, end) 包括 begin，不包括end
   * slice()方法返回一个新的数组对象，这一对象是一个由 begin 和 end 决定的原数组的浅拷贝（包括 begin，不包括end）。原始数组不会被改变。
   * 返回值一个新的数组
  */
  const animals = ['ant', 'bison', 'camel', 'duck', 'elephant'];


  console.log(animals.slice(2));
  // expected output: Array ["camel", "duck", "elephant"]


  console.log(animals.slice(2, 4));
  // expected output: Array ["camel", "duck"]


  console.log(animals.slice(1, 5));
  // expected output: Array ["bison", "camel", "duck", "elephant"]


  /**
   * 13. Array.concat()
   * 此方法是一个可以将多个数组拼接成一个数组：
   * 返回值为合并后新的数组,并且不会去重
  */
  let array = [1, 2]
  let array2 = [2, 3, 4]
  let newArray = array.concat(array2)
  console.log('原数组 array', array);
  console.log('原数组 array2', array2);
  console.log('合并后新的数组', newArray);

  /**
   * 14. Array.isArray()
   * 判断一个对象是否为数组,返回值是Boolean
  */
  let stu = [1, 2, 3, 4];
  let tea = '123'
  console.log('stu是否为数组[1, 2, 3, 4]:', Array.isArray(stu));
  console.log('tea是否为数组123:', Array.isArray(tea));

  /**
   * 15. Array.toString() Array.join()
   * 此方法将数组转为字符串
  */
  let arrToStr = [1, 2, 3, 4];
  let str = arrToStr.toString();
  console.log(str); // 1,2,3,4
  let str2 = arrToStr.join('|');
  console.log(str2);  // 1|2|3|4
